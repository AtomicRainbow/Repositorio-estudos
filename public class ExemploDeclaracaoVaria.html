public class ExemploDeclaracaoVariavel { public static void main(String[] args) { // Declarando variáveis int numeroInteiro; // Variável do tipo inteiro double numeroDecimal; // Variável do tipo double (número decimal) boolean condicao; // Variável do tipo booleano String texto; // Variável do tipo String (cadeia de caracteres) // Inicializando variáveis numeroInteiro = 10; numeroDecimal = 3.14; condicao = true; texto = "Hello, Java!"; // Exibindo valores das variáveis System.out.println("Número Inteiro: " + numeroInteiro); System.out.println("Número Decimal: " + numeroDecimal); System.out.println("Condição: " + condicao); System.out.println("Texto: " + texto); } }
